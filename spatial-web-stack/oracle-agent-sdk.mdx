## Quick Start

### Agent Configuration

<CodeGroup>
  ```typescript Agent Setup
  import { createOracleAgent } from "@ixo/oracle-agent-sdk";

  // Initialize agent
  const agent = await createOracleAgent({
    did: "did:ixo:oracle/123",
    capabilities: ["verification", "conversation"],
    model: "gpt-4"
  });

  // Configure verification rules
  agent.setVerificationRules({
    requiredConfidence: 0.95,
    evidenceTypes: ["telemetry", "location"]
  });
  ```

  ```typescript Matrix Integration
  // Create secure data room
  const room = await agent.createDataRoom({
    name: "Oracle Workspace",
    encryption: true,
    members: ["@user:ixo.world"]
  });

  // Store verification results
  await agent.storeResult(room.roomId, {
    type: "Verification",
    result: verificationResult,
    evidence: proofs
  });
  ```
</CodeGroup>

### Conversation Management

<CodeGroup>
  ```typescript Chat Integration
  import { useOracleChat } from "@ixo/oracle-agent-sdk/react";

  function OracleChat() {
    const { messages, sendMessage, isLoading } = useOracleChat({
      oracleId: "did:ixo:oracle/123",
      sessionId: "session-456"
    });

    return (
      <div>
        {messages.map(msg => (
          <ChatMessage key={msg.id} {...msg} />
        ))}
        <MessageInput onSend={sendMessage} />
      </div>
    );
  }
  ```

  ```typescript Tool Calls
  // Register custom tools
  agent.registerTool("locationVerify", async (params) => {
    const { latitude, longitude } = params;
    const proof = await agent.verifyLocation({
      coordinates: [latitude, longitude],
      timestamp: Date.now()
    });
    return proof;
  });

  // Handle tool responses
  agent.onToolCall(async (call) => {
    if (call.type === "locationVerify") {
      return await agent.tools.locationVerify(call.params);
    }
  });
  ```
</CodeGroup>

### Dynamic UI Components

```typescript
import { OracleUI } from "@ixo/oracle-agent-sdk/ui";

// Register custom components
OracleUI.register("VerificationCard", VerificationCard);
OracleUI.register("LocationProof", LocationProof);

// Render dynamic content
function DynamicContent({ content }) {
  return (
    <OracleUI.Renderer
      content={content}
      components={{
        VerificationCard,
        LocationProof
      }}
    />
  );
}
```

### Verification Flow

```typescript
// Create verification session
const session = await agent.createVerification({
  type: "DataVerification",
  subject: "did:ixo:entity/456",
  context: {
    location: spatialContext,
    timestamp: Date.now()
  }
});

// Process data
const result = await session.verify(data, {
  rules: verificationRules,
  evidence: supportingData
});

// Generate proof
const proof = await session.generateProof({
  result,
  method: "Ed25519Signature2020"
});
```
